-- Stratégie: Schema-per-tenant pour isolation maximale
-- 100K+ tenants → shared schema + RLS pour optimiser

CREATE SCHEMA nexa_shared;

-- Tenant registry (schema public)
CREATE TABLE tenants (
  id          UUID         DEFAULT gen_random_uuid() PRIMARY KEY,
  slug        VARCHAR(50)  UNIQUE NOT NULL,
  plan        VARCHAR(20)  DEFAULT 'starter',  -- starter|pro|enterprise
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- Enable RLS on all tenant tables
ALTER TABLE contacts ENABLE ROW LEVEL SECURITY;
CREATE POLICY tenant_isolation ON contacts
  USING (tenant_id = current_setting('app.tenant_id')::UUID);

-- Set tenant context in each request
SELECT set_config('app.tenant_id', $1, true);

-- Composite indexes for multi-tenant queries
CREATE INDEX idx_contacts_tenant_score
  ON contacts(tenant_id, ai_score DESC);

CREATE INDEX idx_orders_tenant_status_date
  ON orders(tenant_id, status, created_at DESC);

-- Full-text search for contacts (Darija + Arabic + French)
CREATE INDEX idx_contacts_fts
  ON contacts USING GIN(to_tsvector('arabic', first_name || ' ' || last_name));

-- Vector similarity search for AI lead matching
CREATE INDEX idx_contacts_embedding
  ON contacts USING ivfflat (embedding vector_cosine_ops)
  WITH (lists = 100);  -- pgvector

-- TimescaleDB continuous aggregates
CREATE MATERIALIZED VIEW revenue_daily
  WITH (timescaledb.continuous) AS
  SELECT time_bucket('1 day', ts), tenant_id,
         sum(value) AS daily_revenue
  FROM metrics WHERE metric_name = 'revenue'
  GROUP BY 1, 2;

import { createCipheriv, randomBytes } from 'crypto'

class EncryptionService {
  private readonly algo = 'aes-256-gcm'

  encrypt(plaintext: string): EncryptedData {
    const iv  = randomBytes(16)
    const key = Buffer.from(process.env.ENCRYPTION_KEY)
    const cipher = createCipheriv(this.algo, key, iv)
    const encrypted = Buffer.concat([
      cipher.update(plaintext, 'utf8'),
      cipher.final()
    ])
    return { encrypted, iv, tag: cipher.getAuthTag() }
  }
  // Used for: phone numbers, payment data
  // card numbers, personal info (RGPD)
}

TypeScript · AI Orchestrator
class NexaAIOrchestrator {
  /** WhatsApp: Multi-step conversation pipeline */
  async processWaMessage(msg: WaMessage): Promise<AiResponse> {
    // Step 1: Language detection (FastText — 98% accuracy Darija)
    const lang = await this.langDetect.detect(msg.body)

    // Step 2: Retrieve conversation history + tenant context (RAG)
    const context = await this.rag.retrieve({
      tenantId: msg.tenantId, contactId: msg.contactId,
      query: msg.body, topK: 5
    })

    // Step 3: Classify intent (order|quote|info|complaint|appointment)
    const intent = await this.classifier.classify(msg.body, lang)

    // Step 4: Generate response with Claude 3.5
    const response = await this.llm.generate({
      model:  'claude-3-5-sonnet-20241022',
      system: buildSystemPrompt(lang, intent, context),
      messages: [{ role: 'user', content: msg.body }],
      max_tokens: 500
    })

    // Step 5: Post-process (extract order, quote, appointment data)
    const extracted = await this.extractor.extract(response, intent)

    // Step 6: Trigger business actions (create order, update CRM...)
    if (extracted.order)      await this.ecom.createOrder(extracted.order)
    if (extracted.lead)       await this.crm.qualifyLead(extracted.lead)
    if (extracted.appointment) await this.calendar.book(extracted.appointment)

    return { message: response, actions: extracted, lang, intent }
  }
}

name: NexaMarket CI/CD Pipeline

on:
  push:
    branches: [main, develop]

    branches: [main]

  test:
    runs-on: ubuntu-24.04

      postgres: { image: postgres:16, env: {POSTGRES_DB: nexatest} }

      - uses: actions/checkout@v4
      - name: Unit tests (Jest)
        run: pnpm test:unit --coverage --threshold=80
      - name: Integration tests
        run: pnpm test:e2e
      - name: Upload coverage → Codecov
        uses: codecov/codecov-action@v4

  security:
    needs: [test]
    steps:
      - name: SAST — Semgrep
        uses: semgrep/semgrep-action@v1
      - name: Dependency scan — Snyk
        run: snyk test --severity-threshold=high
      - name: Docker image scan — Trivy
        uses: aquasecurity/trivy-action@master

  build-push:
    needs: [security]
    steps:
      - name: Build & push Docker image
        run: |
          docker buildx build --platform linux/amd64,linux/arm64 \
            -t ghcr.io/nexamarket/$SERVICE:${{ github.sha }} \
            --cache-from type=registry,ref=ghcr.io/nexamarket/$SERVICE:cache \
            --push .

  deploy-staging:
    needs: [build-push]
    environment: staging
    steps:
      - name: Deploy to K8s staging
        run: kubectl set image deploy/$SERVICE $SERVICE=ghcr.io/nexamarket/$SERVICE:$SHA
      - name: Smoke tests
        run: pnpm test:smoke --env=staging

  deploy-prod:
    needs: [deploy-staging]
    environment: production    # Requires manual approval
    steps:
      - name: Blue/Green deploy
